## Code Highlighting

```typescript
interface User {
  id: number
  name: string
  email?: string
}

const users: User[] = [
  { id: 1, name: 'John', email: 'john@example.com' },
  { id: 2, name: 'Jane' }
]

const getUser = (id: number): User | undefined =>
  users.find(user => user.id === id)
```
